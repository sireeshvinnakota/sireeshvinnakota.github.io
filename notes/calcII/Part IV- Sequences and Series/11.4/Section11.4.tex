\begin{center}
\textbf{Comparison Tests}
\end{center}

\begin{thm}[Comparison Test]
Suppose that $\sum a_n$ and $\sum b_n$ are series with positive terms. 
\begin{enumerate}[label=$(\roman*)$]
\item If $\sum b_n$ is convergent and $a_n \leq b_n$ for all $n$, then $\sum a_n$ is also convergent.
\item If $\sum b_n$ is divergent and $a_n \geq b_n$ for all $n$, then $\sum a_n$ is also divergent.
\end{enumerate}
\end{thm}

\begin{aun}
As you can see, the above statement has some specific conditions. If, suppose, we're in a case where it's easy to find a ``bigger'' series that diverges, the statement above wouldn't apply and we'd be out of luck; for this reason, we'll need another tool to ``tie'' series together. 
\end{aun}


\begin{thm}[Limit Comparison Test]
Suppose that $\sum a_n$ and $\sum b_n$ are series with positive terms. If
\[
\lim_{n \to \infty} \frac{a_n}{b_n} = c
\]
for some constant finite number $c > 0$, then either both series converge or both diverge. 
\end{thm}

\begin{exx}
Determine whether the following series converge or diverge:
\[
\sum_{k = 1}^\infty \frac{\ln(k)}{k} \mte{ and } \sum_{n = 2}^\infty \frac{n - 1}{\sqrt{n^6 + 1}}
\]
\end{exx}
\ans{Diverges, compare with $\sum \frac1k$. The second converges with comparison to $1/n^2$.}

\begin{exx}
Determine whether the following series converge or diverge:
\[
\sum_{n = 1}^\infty \frac{4n^2 - n}{n^3 + 9}
\]
\end{exx}

\ans{Removing the ``filler'' terms, we have the string of inequalities, we see that:
\[
\frac{4n^2 - n}{n^3 + 9} \leq \frac{4n^2}{n^3 + 9} \leq \frac{4n^2 - n}{n^3} = \frac4n
\]
From here, we see that $\sum \frac4n$ diverges, but since it's a \emph{bigger} sequence that diverges, we cannot use the comparison test. So, defining:
\[
a_n = \frac{4n^2 - n}{n^3 + 9} \mte{ and } b_n = \frac4n
\]
let's complete a \emph{limit} comparison:
\[
\lim_{n \to \infty} \lrabs{\frac{a_n}{b_n}} = \lim_{n \to \infty} \lrabs{\frac{4n^2 - n}{n^3 + 9} \cdot \frac n4} = \lim_{n \to \infty} \lrabs{\frac{n^3 \lrp{4 - \frac{1}{n}}}{n^3 \lrp{4 + \frac{36}{n^3}}}} = 1
\]
Now, by limit comparison, and the fact that:
\[
4\sum_{n = 1}^\infty \frac1n
\]
diverges by $p$-series, we know that the original series diverges as well. 
}




\problembox{Determine whether the following series converges or diverges:
\[
\sum_{n = 1}^\infty \frac{2n^3 + 7}{n^4 \sin^2 (n)}
\]
}
\ans{Notice that $0 \leq \sin^2(n) \leq 1$, so we can write:
\[
\frac{2n^3 + 7}{n^4 \sin^2(n)} \geq \frac{2n^3}{n^4 \sin^2(n)} \geq \frac{2n^3}{n^4} = \frac{2}{n}
\]
and get divergence by comparison. 
}

\problembox{
Determine whether the following series converges or diverges:
\[\sum_{n=1}^{\infty} \frac{\sqrt{2 n^2+4 n+1}}{n^3+9}\]
}
\ans{\textbf{HINT:} Use limit comparison with the series:
\[
\sum_{n = 0}^ \infty \frac{\sqrt{2}}{n^2}
\]
(Similar steps to above show where this comes from.)
}
