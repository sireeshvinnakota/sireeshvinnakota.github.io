\begin{center}
\textbf{Volumes of Rotation}
\end{center}

\begin{aun}
Volumes by washers or disks are an important application of integration, and can be computed using the formulas:
\[
\underbrace{V = \pi \int_a^b \lrb{R(x)}^2 dx}_{\text{Disk}} \mte{and} \underbrace{V = \pi \int_a^b \lrb{R(x)^2 - r(x)^2} dx}_{\text{Washer}}
\]
where $R(x)$ represents the ``radius'' or distance between the function and the axis of rotation, and, in the case of the Washer method, $r(x)$ represents the ``inner radius.''
\end{aun}

\begin{exx}
Find the volume of the solid obtained by rotating the region bounded by the given curves about the specified line. Sketch the region, and state the method of integration before computing the integral:
\[
y = 1-x^2 \mte{ about the } x-\text{axis}
\]
\end{exx}

\ans{\href{https://www.geogebra.org/m/zBRtUVfR}{Graph} This is the disk method, so we have the integral:
\[
V = \pi \int_{-1}^1 \lrp{1-x^2}^2 dx
\]
}


\problembox{
Find the volume of the solid obtained by rotating the region bounded by the given curves about the specified line. Sketch the region, and state the method of integration before computing the integral:
\[
y = 6 - x^2, \quad y = 2 \mte{ about the } x\text{-axis}
\]
}

\ans{
The correct integral to set up is:
\[
V = \pi \int_{-2}^2 (6-x^2)^2 - (2)^2 dx
\]
using the \emph{Washer} method. From here, integration is straightforward.
}

\problembox{Find the volume of the solid obtained by rotating the region bounded by the given curves about the specified line. Sketch the region, and state the method of integration before computing the integral:
\begin{enumerate}[label=$\roman*.$]
\item $y = x^3, \quad y = 0, \quad x = 1 \mte{ about the } x = 2$
\item $x = y^2, \quad x = 1-y^2 \mte{ about the } x=3$
\end{enumerate}
}

\ans{Both need one to convert equations to functions of $y$.
\begin{enumerate}[label=$\roman*.$]
\item Using $x = y^{1/3}$ we see:
\[
V = \pi \int_0^1 (y^{1/3})^2 - 1^2 dy
\]
\item Our bounds of integration here are $y = \pm \sqrt{2}/2$:
\[
V = \pi \int_{-\sqrt{2}/2}^{\sqrt{2}/2} (y^2)^2 - (1-y^2)^2 dy
\]
\end{enumerate}}
