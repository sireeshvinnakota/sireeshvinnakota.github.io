\begin{center}
\textbf{Average Value of a Function}
\end{center}

\begin{prop}
Let $f$ be an integrable function defined on the interval $[a,b]$. Then the average value of $f$, $f_{avg}$ on the defined interval is:
\[
f_{avg} := \frac1{b-a} \int_a^b f(x) dx
\]
\end{prop}
\begin{thm}[Mean Value Theorem for Integrals] If $f$ is a continuous function on the interval $[a,b]$, then there exists $c \in [a,b]$ for which 
\[
f(c) = f_{avg} \mte{ that is } f(c) (b-a) = \int_a^b f(x)dx
\]
\end{thm}
\begin{exx}
Find the average value of the following function on the given interval:
\[
f(x) = \cos^4(x) \sin (x)  \mte{ defined on } [0,\pi]
\]
\end{exx}

\ans{Using the theorem above, we can compute:
\[
f_{avg} = \frac{1}{\pi - 0} \int_0^\pi \cos^4 x \sin(x) dx
\]
We may proceed by $u$-substitution, on $u = \cos (x)$ to find the answer. 
}

\problembox{Compute the average value of the function:
\[
f(x) = 8x - 3 + 4e^{2-x} dx \mte{ on the interval } [0,2]
\]
}
\ans{We set up the same integral as before:
\[
f_{avg} = \frac12\int_{0}^2 8x - 3 + 4e^{2-x}dx
\]
The only twist in the remainder of the problem is the antiderivative of $e^{2-x}$, which is just $-e^{2-x}$.
}

\problembox{[Word Problem] The temperature (in $F$) $t$ hours after 9 a.m. is given by the function:
\[
T(t) := 50 + 14 \sin \lrp{\frac{\pi t}{12}}
\]
Compute the average temperature between 9 a.m. and 9 p.m. using the function above.
}
\ans{The integral to set up is:
\[
T_{avg} = \frac{1}{12} \int_0^{12} T(t) dt
\]}
