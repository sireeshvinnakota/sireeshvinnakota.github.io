\begin{center}
\textbf{The Fundamental Theorem of Calculus}
\end{center}
\begin{thm}[Fundamental Theorem of Calculus]
Suppose that $f$ is a continuous function on the interval $[a,b]$. Then:
\begin{enumerate}[label=$(\roman*)$]
\item If $g(x) = \int_{a}^x f(t)dt$ then $g'(x) = f(x)$.
\item $\int_a^b f(x)dx = F(b) - F(a)$ for \emph{any} function $F(x)$ satisfying $F'(x) = f(x)$.
\end{enumerate}
\end{thm}
\begin{aun}
A common problem in this section follows the form:
\[
\frac{d}{dx} \int_{a(x)}^{b(x)} f(t)dt =  f(b(x))b'(x) - f(a(x))a'(x)
\]
\end{aun}

\begin{exx}
Define the function:
\[
h(x) = \int_1^{e^x} \ln(t) dt,
\]
compute $h'(x)$.
\end{exx}

\ans{Following the author's note above:
\[
\frac{d}{dx} h(x) = \ln(e^x) \cdot e^x + 0 = xe^x
\]}

\problembox{Given the function:
\[
g(x) := \int_0^x \sqrt{t + t^3} dt,
\]
compute the derivative $g'(x)$ and the slope of the line tangent to $g(x)$ at $x = 5$.
}

\ans{$\displaystyle g'(x) = \sqrt{x+x^3}$. Plug in to compute slope of the tangent line.}

\problembox{Compute the derivative of the function:
\[
y (x) := \int_{\sqrt{x}}^{\pi/4}  \theta \tan \theta d\theta
\]}

\ans{\[
y'(x) = - \sqrt{x} \tan\sqrt{x} \cdot \frac{1}{2 \sqrt{x}} = \frac{-1}{2} \tan\sqrt{x}
\]}

\problembox{Evaluate the following integrals using the Fundamental Theorem of Calculus:
\begin{enumerate}[label=$\roman*.$]
\item $\displaystyle \int_{-1}^1 x^{100}$
\item $\displaystyle \int_{\pi/4}^{\pi/3} \csc^2 \theta d \theta$
\end{enumerate}}

\ans{First is $[x^{101}/101]_{-1}^1 = \frac{2}{101}$ and second is $[-\cot(x)]_{\pi/4}^{\pi/3}$.}


\problembox{[Challenge] Evaluate the limit:
\[
\lim_{n \to \infty} \sum_{i = 1}^n \lrp{\frac{i^4}{n^5} + \frac{i}{n^2}}
\]
}

\ans{The trick here is to evaluate this as an integral. First, factor out $1/n$ from the summand:
\[
\frac{i^4}{n^5} + \frac{i}{n^2} = \frac1n \lrp{ \lrp{i/n}^4 + (i/n)}
\]
from here, we see that $a = 0$, $\Delta x = 1/n$ so $b = 1$. We're home free when we write down:
\[
\int_0^1 x^4 + x dx = \lrb{\frac{x^5}{5} + \frac{x^2}{2}}_0^1 = \frac{1}{5} + \frac12 = \frac{7}{10}
\]
}
